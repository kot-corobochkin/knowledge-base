ON list.`request_id`=queue.`request_id` LEFT JOIN `req_processing` AS `proc` 
ON proc.`request_id` = queue.`request_id` AND proc.`type_id` = queue.`type_id` 
LEFT JOIN `req_queue_priority` AS prior ON (prior.`type_id` = queue.`type_id` AND prior.`copy_id` = queue.`copy_id` AND prior.`user_id` = queue.`user_id`) WHERE queue.`type_id` = '114' AND (

            )  AND (queue.`user_id` != '1' OR queue.`copy_id` != '0') 
				AND NOT (queue.copy_id = 4 AND queue.user_id in (51618))  
				AND proc.`from_api` = '0' 
				ORDER BY
