FROM `req_processing` AS p LEFT JOIN `req_list` AS list ON p.`request_id` = list.`request_id` 
LEFT JOIN deltaincom_clients.users u ON u.user_id = list.user_id LEFT JOIN req_types rt ON rt.type_id = p.type_id 
WHERE (list.user_id != 1 OR list.copy_id != 0) AND list.created_time &lt; NOW() - INTERVAL 20 MINUTE ORDER BY p.`created_time` DESC
LIMIT 10</details>
