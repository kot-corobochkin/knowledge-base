<details><summary>Число быстрых и медленных запросов пользователя</summary>SELECT 
    CONCAT(
        DATE_FORMAT(DATE_SUB(created_time, INTERVAL WEEKDAY(created_time) DAY), '%d %b'),
        ' - ',
        DATE_FORMAT(DATE_ADD(DATE_SUB(created_time, INTERVAL WEEKDAY(created_time) DAY), INTERVAL 6 DAY), '%d %b')
    ) AS week_range,

    -- Агролизинг (ваши user_id)
    SUM(CASE 
        WHEN user_id IN (54131, 56495, 57814)
             AND TIMESTAMPDIFF(MINUTE, created_time, processed_time) > 20
        THEN 1 ELSE 0 END) AS agro_slow,

    SUM(CASE 
        WHEN user_id IN (54131, 56495, 57814)
             AND processed_time IS NOT NULL
        THEN 1 ELSE 0 END) AS agro_processed,

    -- Всего
    SUM(CASE 
        WHEN processed_time IS NOT NULL
        THEN 1 ELSE 0 END) AS total_processed,

    SUM(CASE 
        WHEN processed_time IS NULL 
             OR TIMESTAMPDIFF(MINUTE, created_time, processed_time) > 20
        THEN 1 ELSE 0 END) AS total_slow

FROM req_list
WHERE created_time >= NOW() - INTERVAL 3 WEEK
GROUP BY YEARWEEK(created_time, 1)
ORDER BY MIN(created_time);</details>

    cu.user_id,
    cu.login,
    up.period_id,
    up.tarif_id,
    rq.req_type_id,
    rt.name AS type_name,

    rq.*


FROM deltaincom_security.user_periods up

JOIN deltaincom_security.req_quotas rq
    ON rq.period_id = up.period_id

JOIN deltaincom_security.req_types rt
    ON rt.type_id = rq.req_type_id

JOIN deltaincom_clients.users cu
    ON cu.user_id = up.user_id

  AND rq.req_type_id IN (367, 328, 291, 327)
  AND up.end_date > CURDATE();</details>
