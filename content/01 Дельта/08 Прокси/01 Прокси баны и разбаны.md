```sql
SELECT 
    ph.host AS source,

    COUNT(DISTINCT l.id) AS total_proxies,

    COUNT(DISTINCT CASE 
        WHEN b.used = 0 OR b.ban_time > NOW()
        THEN l.id 
    END) AS banned_proxies,

    COUNT(DISTINCT CASE 
        WHEN (b.used = 1 OR b.used IS NULL)
         AND (b.ban_time <= NOW() OR b.ban_time IS NULL)
        THEN l.id 
    END) AS unbanned_proxies

FROM proxy_list l

JOIN proxy_hosts ph
    ON 1=1   -- ⬅️ берём ВСЕ хосты

LEFT JOIN proxy_host_banned b
    ON b.proxy_id = l.id
    AND b.host_id = ph.id

WHERE l.working = 1
AND l.group_id = 0

GROUP BY ph.id, ph.host
ORDER BY total_proxies DESC;
```
